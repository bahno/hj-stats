# hj-stats v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a static React app that calculates World Athletics high-jump ranking points (Performance + Placing) and visualizes how the score differs across competition categories, deployed to GitHub Pages.

**Architecture:** Pure-TypeScript scoring engine (data injected as arguments) consumed by two React views (Calculator, Compare). Reference data lives as committed JSON produced by a Python pipeline (Scoring Tables PDF → performance points; ranking-rules HTML → placing points), verified against known values and refreshed weekly via an auto-PR Action. Everything the user touches is static; GitHub Actions builds Vite and deploys to Pages.

**Tech Stack:** React 18, Vite, TypeScript, Vitest + React Testing Library, Recharts, Python 3 (pdfplumber, requests, beautifulsoup4), GitHub Actions.

## Global Constraints

- Repo: `bahno/hj-stats` (public), deployed to GitHub Pages at `https://bahno.github.io/hj-stats/`.
- Vite `base` MUST be `'/hj-stats/'` so assets resolve under the project subpath.
- Competition categories, in order: `OW, DF, GW, GL, A, B, C, D, E, F`.
- Genders: `men`, `women` (high jump uses different scoring tables per gender).
- Scoring model (single result): `Result Score = Performance Score + Placing Score`.
- Performance points are looked up exactly from the scoring table (no interpolation); inputs are constrained to marks present in the table.
- Engine functions are PURE and receive data as arguments (no direct JSON imports inside the engine) so they unit-test against fixtures.
- Data JSON is committed and human-reviewable; the pipeline never auto-merges — it opens PRs.
- Known verification values (placing, 1st place, T&F final): OW 375, DF 240, GW 200, GL 170, A 140.

```ts
import type { CategoryCode, Gender, PlacingPoints, ScoringTable } from '../data/types';

/** Normalizes a height in metres to the table key format, e.g. 2.3 -> "2.30". */
export function markKey(heightMeters: number): string {
  return heightMeters.toFixed(2);
}

export function performanceScore(
  table: ScoringTable,
  gender: Gender,
  heightMeters: number,
): number {
  const key = markKey(heightMeters);
  const points = table.points_by_mark[gender][key];
  if (points === undefined) {
    throw new Error(`No scoring-table entry for ${gender} high jump at ${key} m`);
  }
  return points;
}

export function placingScore(
  placing: PlacingPoints,
  category: CategoryCode,
  position: number,
): number {
  return placing.final[category][String(position)] ?? 0;
}
```

```ts
export interface ScoreBreakdown {
  performance: number;
  placing: number;
  total: number;
}

export function resultScore(
  table: ScoringTable,
  placing: PlacingPoints,
  gender: Gender,
  heightMeters: number,
  position: number,
  category: CategoryCode,
): ScoreBreakdown {
  const performance = performanceScore(table, gender, heightMeters);
  const placingPts = placingScore(placing, category, position);
  return { performance, placing: placingPts, total: performance + placingPts };
}

export interface CategoryScore extends ScoreBreakdown {
  category: CategoryCode;
}

export function compareCategories(
  table: ScoringTable,
  placing: PlacingPoints,
  gender: Gender,
  heightMeters: number,
  position: number,
  categories: CategoryCode[],
): CategoryScore[] {
  return categories.map((category) => ({
    category,
    ...resultScore(table, placing, gender, heightMeters, position, category),
  }));
}
```

```tsx
      </select>
    </label>
  );
```

```tsx
function ordinal(n: number): string {
  const s = ['th', 'st', 'nd', 'rd'];
  const v = n % 100;
  return n + (s[(v - 20) % 10] ?? s[v] ?? s[0]);
}
```

```python
"""Scrape World Athletics placing points (T&F final) into src/data/placing_points.json."""
import json, re, sys
from pathlib import Path
import requests
from bs4 import BeautifulSoup

URL = "https://worldathletics.org/world-ranking-rules/track-field-events-2025"
CATEGORIES = ["OW", "DF", "GW", "GL", "A", "B", "C", "D", "E", "F"]
OUT = Path(__file__).resolve().parent.parent / "src" / "data" / "placing_points.json"

def fetch() -> BeautifulSoup:
    resp = requests.get(URL, headers={"User-Agent": "hj-stats-pipeline"}, timeout=30)
    resp.raise_for_status()
    return BeautifulSoup(resp.text, "html.parser")

def find_final_table(soup: BeautifulSoup):
    """Return the placing table whose header row contains the category codes."""
    for table in soup.find_all("table"):
        headers = [th.get_text(strip=True) for th in table.find_all("th")]
        if all(cat in headers for cat in ("OW", "DF", "GW", "GL", "A")):
            return table
    raise SystemExit("Could not locate the placing table by category headers")

def parse(table) -> dict:
    headers = [th.get_text(strip=True) for th in table.find_all("th")]
    col = {cat: headers.index(cat) for cat in CATEGORIES if cat in headers}
    final = {cat: {} for cat in CATEGORIES}
    for row in table.find("tbody").find_all("tr"):
        cells = [td.get_text(strip=True) for td in row.find_all(["td", "th"])]
        m = re.match(r"(\d+)", cells[0])
        if not m:
            continue
        position = m.group(1)
        for cat, idx in col.items():
            val = cells[idx].replace(",", "")
            if val.isdigit():
                final[cat][position] = int(val)
    return final

def main():
    final = parse(find_final_table(fetch()))
    OUT.write_text(json.dumps({
        "source": "World Athletics World Ranking Rules 2025 — Track & Field (final)",
        "final": final,
    }, indent=2) + "\n")
    print(f"Wrote {OUT}")

if __name__ == "__main__":
    main()
```
